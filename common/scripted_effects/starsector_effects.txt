starsector_give_colony_effect = {
	# ROOT scope is country
	log = "COUNTRY [root.GetName]: Adding colony."
	#Scope to home system first and add colonies
	owner.capital_scope.solar_system = {
		every_system_planet = {
			limit = {
				is_colony = no
				is_colonizable = yes
			}
			save_event_target_as = original_planet
			create_colony = {
				owner = root
				species = owner_main_species
			}
			set_owner = root
			set_controller = root
			#set_name = "[event_target:original_planet.GetName]"
			set_name = "heg_aztlan_system_initializer_COATL"
			log = "Created COUNTRY [root.GetName] COLONY [this.GetName] in [prev.GetName] system"
		}
	}
}
